quantify_categorical <- function(x) {
    # NA - Non-Existent/Terrible/etc - 0
    x[x == 'NA'] <- 0
    # Po - Poor - 1
    x[x == 'Po'] <- 1
    # Fa - Fair - 2
    x[x == 'Fa'] <- 2
    # TA - Typical/Average - 3
    x[x == 'TA'] <- 3
    # Gd - Good - 4
    x[x == 'Gd'] <- 4
    # Ex - Excellent - 5
    x[x == 'Ex'] <- 5
    # Su - Superior - 6
    x[x == 'Su'] <- 6
    return(as.numeric(x))
}
